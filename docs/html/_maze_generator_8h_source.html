15// Directions: 0 = North, 1 = East, 2 = South, 3 = West.
16enum Direction { UP = 0, RIGHT = 1, DOWN = 2, LEFT = 3 };
17
18struct Cell {
19 bool visited;
20 bool walls[4]; // walls[0]=North, [1]=East, [2]=South, [3]=West.
21 Cell() : visited(false) {
22 walls[0] = walls[1] = walls[2] = walls[3] = true;
23 }
24};
25
26typedef std::vector<std::vector<Cell>> Maze;
27
28class MazeGenerator {
29public:
30 MazeGenerator(int rows, int cols)
31 : rows(rows), cols(cols), finished(false) {
32 maze.resize(rows, std::vector<Cell>(cols));
33 }
34 virtual ~MazeGenerator() {}
35 // Perform one generation step.
36 virtual bool step() = 0;
37 bool isFinished() const { return finished; }
38 // Return the maze (2D grid of Cells).
39 const std::vector<std::vector<Cell>>& getMaze() const { return maze; }
40 // Non-const version for modification.
41 std::vector<std::vector<Cell>>& getMazeRef() { return maze; }
42 // Optionally, return the current cell (for visualization).
43 virtual QPoint getCurrentCell() const { return QPoint(-1, -1); }
44protected:
45 int rows, cols;
46 bool finished;
47 std::vector<std::vector<Cell>> maze;
48
49};
